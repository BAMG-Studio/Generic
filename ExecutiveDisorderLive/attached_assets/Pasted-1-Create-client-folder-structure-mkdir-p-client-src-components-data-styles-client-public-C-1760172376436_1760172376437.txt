  return (
    <div className="app-shell layer-ui">
      <div className="layer-bg visual-bg">
        <h1 className="text-4xl font-bold text-center py-8">Executive Disorder</h1>

        >
          Count: {count}
        </button>
      </div>
    </div>
  );
}
